> /// Process items through multi-turn CAP scheduling.
> /// Items become ready at their base_turn. Each turn, each lane can accept up to `cap` items.
> /// Overflow items remain for subsequent turns.
> /// Returns items in acceptance order: by accepted turn, then lane name, then plan-file seq.
> pub fn fold_b(a: &[R0], cap: usize) -> Vec<R0> {
>     let cap = cap.max(1);
>
>     // Find max base_turn to know where items first become ready
>     let max_base = a.iter().map(|r| r.base_turn).max().unwrap_or(0);
>
>     // Collect all items, grouped by lane
>     let mut pending_by_lane: BTreeMap<String, Vec<R0>> = BTreeMap::new();
>     for row in a.iter() {
>         pending_by_lane.entry(row.lane_name.clone()).or_default().push(row.clone());
>     }
>     // Sort each lane's pending by seq (plan-file order)
>     for items in pending_by_lane.values_mut() {
>         items.sort_by_key(|r| r.seq);
>     }
>
>     let mut result: Vec<R0> = Vec::new();
>     let mut accepted: std::collections::BTreeSet<String> = std::collections::BTreeSet::new();
>
>     // Process turn by turn
>     let mut turn = 0u64;
>     loop {
>         let mut accepted_this_round = false;
>
>         for (_lane, items) in pending_by_lane.iter() {
>             // Count how many ready and not-yet-accepted items exist in this lane for this turn
>             let ready_count = items.iter()
>                 .filter(|r| r.base_turn <= turn && !accepted.contains(&r.task_name))
>                 .count();
>             if ready_count > 0 {
>                 accepted_this_round = true;
>                 break;
>             }
>         }
>
>         if !accepted_this_round && turn > max_base {
>             break;
>         }
>
>         // For each lane (in alphabetical order due to BTreeMap), accept up to cap items
>         let lanes: Vec<String> = pending_by_lane.keys().cloned().collect();
>         for lane in &lanes {
>             let items = pending_by_lane.get(lane).unwrap();
>             let mut count = 0;
>             for row in items.iter() {
>                 if count >= cap {
>                     break;
>                 }
>                 if row.base_turn <= turn && !accepted.contains(&row.task_name) {
>                     let mut accepted_row = row.clone();
>                     accepted_row.turn = turn;
>                     result.push(accepted_row);
>                     accepted.insert(row.task_name.clone());

>                 }
>             }
>         }

>         }
