> /// Process ready items with CAP overflow across turns.
> /// CAP limits how many ready records each lane may accept in one turn;
> /// overflow remains live for later turns.
> pub fn fold_b(a: &[R0], cap: usize) -> Vec<R0> {
>     let cap = cap.max(1);
>
>     // Find all distinct turns
>     let mut all_turns: Vec<u64> = a.iter().map(|r| r.base_turn).collect();
>     all_turns.sort();
>     all_turns.dedup();
>
>     // Overflow pool: items not yet accepted, keyed by lane
>     // Each item in the pool retains its seq for ordering
>     let mut overflow: BTreeMap<String, Vec<R0>> = BTreeMap::new();
>     let mut accepted: Vec<R0> = Vec::new();
>
>     for &turn in &all_turns {
>         // Add items arriving at this turn to the overflow pool
>         for row in a.iter() {
>             if row.base_turn == turn {
>                 overflow.entry(row.lane_name.clone()).or_default().push(row.clone());
>             }
>         }
>
>         // For each lane, accept up to CAP items (in plan-file order = seq order)
>         for (_lane, pool) in overflow.iter_mut() {
>             pool.sort_by_key(|r| r.seq);
>             let take = cap.min(pool.len());
>             let mut taken: Vec<R0> = pool.drain(..take).collect();
>             // Set accepted turn
>             for item in taken.iter_mut() {
>                 item.turn = turn;
>             }
>             accepted.extend(taken);
>         }
>
>         // Remove empty lanes
>         overflow.retain(|_, v| !v.is_empty());
>     }
>
>     // Handle remaining overflow - they get accepted at turn max_turn + 1, +2, etc.
>     // Keep processing until overflow is empty
>     let mut extra_turn = all_turns.last().copied().unwrap_or(0) + 1;
>     while !overflow.is_empty() {
>         for (_lane, pool) in overflow.iter_mut() {
>             pool.sort_by_key(|r| r.seq);
>             let take = cap.min(pool.len());
>             let mut taken: Vec<R0> = pool.drain(..take).collect();
>             for item in taken.iter_mut() {
>                 item.turn = extra_turn;
>             }
