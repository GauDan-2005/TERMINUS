use std::collections::{BTreeMap, BTreeSet};

use crate::core_a::R0;

/// Implements multi-turn CAP acceptance.
/// Each item has a base_turn (arrival turn). At each turn t, items with base_turn <= t
/// that haven't been accepted yet are ready. Within each lane, ready items are sorted
/// by seq (plan-file order). Up to `cap` items per lane are accepted per turn.
/// Overflow remains live for later turns.
pub fn fold_b(a: &[R0], cap: usize) -> Vec<R0> {
    let cap = cap.max(1);

    if a.is_empty() {
        return Vec::new();
    }

    let mut accepted: Vec<R0> = Vec::new();
    let mut accepted_set: BTreeSet<usize> = BTreeSet::new(); // track by index in `a`

    // Find the range of turns we need to process
    let min_turn = a.iter().map(|r| r.base_turn).min().unwrap_or(0);
    let max_base_turn = a.iter().map(|r| r.base_turn).max().unwrap_or(0);

    let mut current_turn = min_turn;

    loop {
        // Collect ready items: base_turn <= current_turn and not yet accepted
        let mut ready_by_lane: BTreeMap<String, Vec<(usize, &R0)>> = BTreeMap::new();
        for (idx, row) in a.iter().enumerate() {
            if row.base_turn <= current_turn && !accepted_set.contains(&idx) {
                ready_by_lane.entry(row.lane_name.clone()).or_default().push((idx, row));
            }
        }

        if ready_by_lane.is_empty() {
            break;
        }

        let mut accepted_this_turn: Vec<(usize, R0)> = Vec::new();

        for (_lane, mut items) in ready_by_lane {
            // Sort by seq (plan-file order)
            items.sort_by_key(|(_, row)| row.seq);
            for (idx, row) in items.into_iter().take(cap) {
                let mut accepted_row = row.clone();
                accepted_row.turn = current_turn;
                accepted_this_turn.push((idx, accepted_row));
            }
        }

        if accepted_this_turn.is_empty() {
            // No progress possible at this turn, try next
            if current_turn > max_base_turn && accepted_set.len() == a.len() {
                break;
            }
            current_turn += 1;
            if current_turn > max_base_turn + a.len() as u64 {
                break; // Safety: prevent infinite loop
            }
            continue;
        }

        // Sort within this turn: by lane name, then by seq
        accepted_this_turn.sort_by(|(_, left), (_, right)| {
            left.lane_name.cmp(&right.lane_name).then(left.seq.cmp(&right.seq))
        });

        for (idx, row) in accepted_this_turn {
            accepted_set.insert(idx);
            accepted.push(row);
        }

        // If all items are accepted, we're done
        if accepted_set.len() == a.len() {
            break;
        }

        current_turn += 1;
        if current_turn > max_base_turn + a.len() as u64 {
            break; // Safety: prevent infinite loop
        }
    }

    accepted
}
