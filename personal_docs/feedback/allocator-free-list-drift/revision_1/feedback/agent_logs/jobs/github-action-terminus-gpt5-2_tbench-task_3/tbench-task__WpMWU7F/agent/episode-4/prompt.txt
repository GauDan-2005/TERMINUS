size_t arena_heap_sig(const Arena *a) {
    size_t sig = 0;
    size_t off = 0;
    while (off < a->used) {
        Block b;

    }

size_t arena_fl_sig(const Arena *a) {
    size_t sig = 0;
    size_t cur = a->fl_head.off;
    while (cur != FL_END) {
        Block b = {(unsigned char *)a->mem, cur};
        sig += tag_read_size(&b);
        cur = a->fl_link_next[fl_ix(cur)];
    }
    return sig;
}

    size_t cur = a->fl_head.off;

        cur = a->fl_link_next[fl_ix(cur)];
    }

}
