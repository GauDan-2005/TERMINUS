import (
"encoding/json"

"reflect"

type entry struct {
Case      string `json:"case"`

Trace     []int  `json:"slot_trace"`

Reuse     bool   `json:"reuse"`
}

type rawDoc struct {

}

type finalDoc struct {

}

func matchEntry(a, b entry) bool {
if a.Case != b.Case {
return false

return false
}
if a.Grouped != b.Grouped {
return false
}
if a.Reuse != b.Reuse {
return false
}
return true
}
