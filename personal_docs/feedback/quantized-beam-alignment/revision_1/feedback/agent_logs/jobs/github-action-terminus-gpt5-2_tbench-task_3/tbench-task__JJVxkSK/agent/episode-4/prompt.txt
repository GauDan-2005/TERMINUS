type rawDoc struct {

type finalDoc struct {
    Rows    []entry        `json:"rows"`
    Summary map[string]any `json:"summary"`
}

func fold(a []entry) []entry {
    seen := map[byte]entry{}
    for _, row := range a {
        key := row.Case[0]
        seen[key] = row
    }
    out := make([]entry, 0, len(seen))
    for _, row := range seen {
        out = append(out, row)
    }
    sort.Slice(out, func(i, j int) bool { return out[i].Case < out[j].Case })
    return out
}

func cx(handle int, row []int) int {
        acc := 0

        }
        return acc

func token(sum int, slot int, packed bool, group bool, reuse bool) int {
        v := sum + slot*7
        if packed {
                v += 3

        if group {
                v += 5
        }
        if reuse {

func runOne(c Case) RawEntry {
        moved := bx(len(c.Rows), c.Seq)
        baseRows := make([][]int, len(c.Rows))
        for i, row := range c.Rows {
                if c.Packed {
                        baseRows[i] = ax(row, i)
                } else {
                        cp := make([]int, len(row))
                        copy(cp, row)
                        baseRows[i] = cp
                }
        }

        vals := make([]int, 0, len(c.Seq))
        trace := make([]int, 0, len(c.Seq))
        for pos, src := range c.Seq {
                h := moved[pos]
                acc := cx(h, baseRows[src])
                vals = append(vals, token(acc, h, c.Packed, c.Group, c.Reuse))
                trace = append(trace, h)
        }

        orderUsed := make([]int, len(c.Seq))
        copy(orderUsed, c.Seq)
        return RawEntry{
                Case:      c.Name,
                Produced:  vals,
                SlotTrace: trace,
                OrderUsed: orderUsed,
                Packed:    c.Packed,
                Grouped:   c.Group,
                Reuse:     c.Reuse,
        }
}
