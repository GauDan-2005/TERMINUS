package main

import (
        "bytes"
        "encoding/json"
        "errors"
        "os"
        "path/filepath"
        "sort"
)

type entry struct {
        Case      string `json:"case"`
        Produced  []int  `json:"produced"`
        Trace     []int  `json:"slot_trace"`
        OrderUsed []int  `json:"order_used"`
        Packed    bool   `json:"packed"`
        Grouped   bool   `json:"grouped"`
        Reuse     bool   `json:"reuse"`
}

type rawDoc struct {
        Raw []entry `json:"raw"`
}

type finalDoc struct {
        Rows    []entry        `json:"rows"`
        Summary map[string]any `json:"summary"`
}

func readJSONFile[T any](path string, out *T) error {
        b, err := os.ReadFile(path)
        if err != nil {
                return err
        }
        dec := json.NewDecoder(bytes.NewReader(b))
        dec.DisallowUnknownFields()
        return dec.Decode(out)
}

func toMap(rows []entry) (map[string]entry, error) {
        m := make(map[string]entry, len(rows))
        for _, r := range rows {
                if r.Case == "" {
                        return nil, errors.New("row missing case")
                }
                if _, ok := m[r.Case]; ok {
                        return nil, errors.New("duplicate case: " + r.Case)
                }
                m[r.Case] = r
        }
        return m, nil
}

func eqSlice(a, b []int) bool {
        if len(a) != len(b) {
                return false
        }
        for i := range a {
                if a[i] != b[i] {
                        return false
                }
        }
        return true
}

// sameSix compares the six required fields, keyed by case name.
func sameSix(a, b entry) bool {
        return a.Case == b.Case &&
                eqSlice(a.Produced, b.Produced) &&
                eqSlice(a.Trace, b.Trace) &&
                eqSlice(a.OrderUsed, b.OrderUsed) &&
                a.Packed == b.Packed &&
                a.Grouped == b.Grouped &&
                a.Reuse == b.Reuse
}

func main() {
        if len(os.Args) != 3 {
                panic("usage: align raw final")
        }
        rawPath := os.Args[1]
        finalPath := os.Args[2]

        // Load incoming raw.
        var raw rawDoc
        if err := readJSONFile(rawPath, &raw); err != nil {
                panic(err)
        }
        rawMap, err := toMap(raw.Raw)
        if err != nil {
                panic(err)
        }

        // Produce one report row per raw row, sorted. (No folding/dropping.)
        rows := make([]entry, 0, len(raw.Raw))
        rows = append(rows, raw.Raw...)
        sort.Slice(rows, func(i, j int) bool { return rows[i].Case < rows[j].Case })

        agree := true

        // Compare new report rows vs incoming raw rows.
        newMap, err := toMap(rows)
        if err != nil {
                agree = false
        } else {
                if len(newMap) != len(rawMap) {
                        agree = false
                }
                for name, rr := range rawMap {
                        nr, ok := newMap[name]
                        if !ok || !sameSix(nr, rr) {
                                agree = false
                                break
                        }
                }
        }

        // If replacing an existing report.json, compare prior report rows to newly
        // generated report rows across those six fields.
        if _, err := os.Stat(finalPath); err == nil {
                var prior finalDoc
                if err := readJSONFile(finalPath, &prior); err != nil {
                        agree = false
                } else {
                        priorMap, err := toMap(prior.Rows)
                        if err != nil {
                                agree = false
                        } else if len(priorMap) != len(newMap) {
                                agree = false
                        } else {
                                for name, pr := range priorMap {
                                        nr, ok := newMap[name]
                                        if !ok || !sameSix(pr, nr) {
                                                agree = false
                                                break
                                        }
                                }
                        }
                }
        }

        doc := finalDoc{Rows: rows, Summary: map[string]any{"total": len(rows), "agree": agree}}
        out, err := json.MarshalIndent(doc, "", "  ")
        if err != nil {
                panic(err)
        }
        if err := os.MkdirAll(filepath.Dir(finalPath), 0o755); err != nil {
                panic(err)
        }
        if err := os.WriteFile(finalPath, append(out, '\n'), 0o644); err != nil {
                panic(err)
        }
}
