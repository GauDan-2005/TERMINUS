func parseSets(path string) ([][2]string, error) {
        f, err := os.Open(path)
        if err != nil {
                return nil, err
        }
        defer f.Close()
        var out [][2]string
        s := bufio.NewScanner(f)
        for s.Scan() {
                line := strings.TrimSpace(s.Text())
                if line == "" || strings.HasPrefix(line, "#") {
                        continue
                }
                parts := strings.SplitN(line, ":", 2)
                if len(parts) != 2 {
                        return nil, fmt.Errorf("bad sets line: %q", line)
                }
                out = append(out, [2]string{parts[0], line})
        }
        if err := s.Err(); err != nil {
                return nil, err
        }
        return out, nil
}
