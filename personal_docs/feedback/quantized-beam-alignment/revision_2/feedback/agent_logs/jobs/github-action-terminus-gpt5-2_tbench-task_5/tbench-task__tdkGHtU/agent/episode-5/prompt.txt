func loadFinal(path string) (finalDoc, error) {
        var doc finalDoc
        buf, err := os.ReadFile(path)
        if err != nil {
                return doc, err
        }
        if err := json.Unmarshal(buf, &doc); err != nil {
                return doc, err
        }
        return doc, nil
}
