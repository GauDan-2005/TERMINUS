package main

import (
"encoding/json"
"os"
"reflect"
)

type entry struct {
Case      string `json:"case"`
Produced  []int  `json:"produced"`
Trace     []int  `json:"slot_trace"`
OrderUsed []int  `json:"order_used"`
Packed    bool   `json:"packed"`
Grouped   bool   `json:"grouped"`
Reuse     bool   `json:"reuse"`
}

type rawDoc struct {
Raw []entry `json:"raw"`
}

type finalDoc struct {
Rows    []entry        `json:"rows"`
Summary map[string]any `json:"summary"`
}

func matchEntry(a, b entry) bool {
if a.Case != b.Case {
return false
}
if !reflect.DeepEqual(a.Produced, b.Produced) {
return false
}
if !reflect.DeepEqual(a.Trace, b.Trace) {
return false
}
if !reflect.DeepEqual(a.OrderUsed, b.OrderUsed) {
return false
}
if a.Packed != b.Packed {
return false
}
if a.Grouped != b.Grouped {
return false
}
if a.Reuse != b.Reuse {
return false
}
return true
}

func main() {
if len(os.Args) != 3 {
panic("usage: align raw final")
}
buf, err := os.ReadFile(os.Args[1])
if err != nil {
panic(err)
}
var raw rawDoc
if err := json.Unmarshal(buf, &raw); err != nil {
panic(err)
}

reportRows := make([]entry, len(raw.Raw))
copy(reportRows, raw.Raw)

agree := true

if oldBuf, err := os.ReadFile(os.Args[2]); err == nil {
var oldDoc finalDoc
if err := json.Unmarshal(oldBuf, &oldDoc); err == nil {
oldByCase := map[string]entry{}
for _, row := range oldDoc.Rows {
oldByCase[row.Case] = row
}
if len(oldDoc.Rows) != len(raw.Raw) {
agree = false
} else {
for _, rawRow := range raw.Raw {
oldRow, exists := oldByCase[rawRow.Case]
if !exists || !matchEntry(rawRow, oldRow) {
agree = false
break
}
}
}
} else {
agree = false
}
}

doc := finalDoc{
Rows:    reportRows,
Summary: map[string]any{"total": len(reportRows), "agree": agree},
}
out, err := json.MarshalIndent(doc, "", "  ")
if err != nil {
panic(err)
}
if err := os.WriteFile(os.Args[2], append(out, 10), 0o644); err != nil {
panic(err)
}
}
