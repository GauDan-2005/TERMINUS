import (
"encoding/json"
"os"
"reflect"
)

type entry struct {
Case      string `json:"case"`
Produced  []int  `json:"produced"`
Trace     []int  `json:"slot_trace"`
OrderUsed []int  `json:"order_used"`
Packed    bool   `json:"packed"`

Reuse     bool   `json:"reuse"`
}

type rawDoc struct {

}

type finalDoc struct {

}

return false

return false
}
if a.Grouped != b.Grouped {
return false
}
if a.Reuse != b.Reuse {
return false
}
return true
}
