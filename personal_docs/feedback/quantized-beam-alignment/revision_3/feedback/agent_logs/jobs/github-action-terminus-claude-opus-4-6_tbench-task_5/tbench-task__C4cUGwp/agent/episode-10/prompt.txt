package main

import (
"encoding/json"
"os"
"reflect"
)

type entry struct {
Case      string `json:"case"`
Produced  []int  `json:"produced"`
Trace     []int  `json:"slot_trace"`
OrderUsed []int  `json:"order_used"`
Packed    bool   `json:"packed"`
Grouped   bool   `json:"grouped"`
Reuse     bool   `json:"reuse"`
}

type rawDoc struct {
Raw []entry `json:"raw"`
}

type finalDoc struct {
Rows    []entry        `json:"rows"`
Summary map[string]any `json:"summary"`
}

func entriesMatch(a, b entry) bool {
if a.Case != b.Case {
return false
}
if !reflect.DeepEqual(a.Produced, b.Produced) {
return false
}
if !reflect.DeepEqual(a.Trace, b.Trace) {
return false
}
if !reflect.DeepEqual(a.OrderUsed, b.OrderUsed) {
return false
}
if a.Packed != b.Packed {
return false
}
if a.Grouped != b.Grouped {
return false
}
if a.Reuse != b.Reuse {
return false
}
return true
}

func computeAgree(rawRows []entry, priorRows []entry) bool {
// Build maps by case name for both
rawMap := make(map[string]entry)
for _, r := range rawRows {
rawMap[r.Case] = r
}
priorMap := make(map[string]entry)
for _, r := range priorRows {
priorMap[r.Case] = r
}
// Check all cases in raw exist in prior and match
if len(rawMap) != len(priorMap) {
return false
}
for name, rawEntry := range rawMap {
priorEntry, ok := priorMap[name]
if !ok {
return false
}
if !entriesMatch(rawEntry, priorEntry) {
return false
}
}
return true
}
