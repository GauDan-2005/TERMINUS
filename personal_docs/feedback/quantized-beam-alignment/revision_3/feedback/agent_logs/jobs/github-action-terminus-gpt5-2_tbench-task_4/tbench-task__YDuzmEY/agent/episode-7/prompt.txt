type entry struct {
        Case      string `json:"case"`
        Produced  []int  `json:"produced"`
        Trace     []int  `json:"slot_trace"`
        OrderUsed []int  `json:"order_used"`
        Packed    bool   `json:"packed"`
        Grouped   bool   `json:"grouped"`
        Reuse     bool   `json:"reuse"`
}

type rawDoc struct {
        Raw []entry `json:"raw"`
}

type finalDoc struct {
        Rows    []entry        `json:"rows"`
        Summary map[string]any `json:"summary"`
}

func readBuiltins(path string) ([]string, error) {
        b, err := os.ReadFile(path)
        if err != nil {
                return nil, err
        }
        lines := strings.Split(string(b), "\n")
        out := make([]string, 0, len(lines))
        for _, ln := range lines {
                ln = strings.TrimSpace(ln)
                if ln == "" {
                        continue
                }
                out = append(out, ln)
        }
        return out, nil
}

func norm(e entry) entry {
        // normalize nil vs empty slices for stable comparisons
        if e.Produced == nil {
                e.Produced = []int{}
        }
        if e.Trace == nil {
                e.Trace = []int{}
        }
        if e.OrderUsed == nil {
                e.OrderUsed = []int{}
        }
        return e
}

func sameSix(a, b entry) bool {
        a = norm(a)
        b = norm(b)
        return reflect.DeepEqual(a.Produced, b.Produced) &&
                reflect.DeepEqual(a.Trace, b.Trace) &&
                reflect.DeepEqual(a.OrderUsed, b.OrderUsed) &&
                a.Packed == b.Packed &&
                a.Grouped == b.Grouped &&
                a.Reuse == b.Reuse
}

func mapByCase(rows []entry) (map[string]entry, map[string]int) {
        m := map[string]entry{}
        cnt := map[string]int{}
        for _, r := range rows {
                cnt[r.Case]++
                m[r.Case] = r // last wins for determinism
        }
        return m, cnt
}

func loadReport(path string) (*finalDoc, error) {
        b, err := os.ReadFile(path)
        if err != nil {
                return nil, err
        }
        var d finalDoc
        dec := json.NewDecoder(bytes.NewReader(b))
        dec.DisallowUnknownFields()
        if err := dec.Decode(&d); err != nil {
                return nil, err
        }
        return &d, nil
}
