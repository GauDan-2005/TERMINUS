package main

import (
        "encoding/json"
        "errors"
        "os"
        "sort"
)

type entry struct {
        Case      string `json:"case"`
        Produced  []int  `json:"produced"`
        Trace     []int  `json:"slot_trace"`
        OrderUsed []int  `json:"order_used"`
        Packed    bool   `json:"packed"`
        Grouped   bool   `json:"grouped"`
        Reuse     bool   `json:"reuse"`
}

type rawDoc struct {
        Raw []entry `json:"raw"`
}

type finalDoc struct {
        Rows    []entry        `json:"rows"`
        Summary map[string]any `json:"summary"`
}

// keep one row per case name (full string). If duplicates exist, last wins.
func byCase(rows []entry) map[string]entry {
        m := make(map[string]entry, len(rows))
        for _, r := range rows {
                m[r.Case] = r
        }
        return m
}

func sortedRows(m map[string]entry) []entry {
        out := make([]entry, 0, len(m))
        for _, r := range m {
                out = append(out, r)
        }
        sort.Slice(out, func(i, j int) bool { return out[i].Case < out[j].Case })
        return out
}

func sameFields(a, b entry) bool {
        if a.Case != b.Case || a.Packed != b.Packed || a.Grouped != b.Grouped || a.Reuse != b.Reuse {
                return false
        }
        if len(a.Produced) != len(b.Produced) || len(a.Trace) != len(b.Trace) || len(a.OrderUsed) != len(b.OrderUsed) {
                return false
        }
        for i := range a.Produced {
                if a.Produced[i] != b.Produced[i] {
                        return false
                }
        }
        for i := range a.Trace {
                if a.Trace[i] != b.Trace[i] {
                        return false
                }
        }
        for i := range a.OrderUsed {
                if a.OrderUsed[i] != b.OrderUsed[i] {
                        return false
                }
        }
        return true
}

func loadPrior(path string) (map[string]entry, bool, error) {
        buf, err := os.ReadFile(path)
        if err != nil {
                if errors.Is(err, os.ErrNotExist) {
                        return nil, false, nil
                }
                return nil, false, err
        }
        var doc finalDoc
        if err := json.Unmarshal(buf, &doc); err != nil {
                return nil, true, err
        }
        return byCase(doc.Rows), true, nil
}
