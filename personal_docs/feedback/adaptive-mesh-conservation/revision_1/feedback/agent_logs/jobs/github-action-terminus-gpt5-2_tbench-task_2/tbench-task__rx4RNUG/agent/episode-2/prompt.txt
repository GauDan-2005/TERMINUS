static int compute_adjacency(const SimState *st) {
    int count = 0;
    for (int i = 0; i < st->grid.n; i++) {
        int p = st->topo.parent[i];
        if (p < 0) continue;
        int c0 = st->topo.child0[p];
        int cn = st->topo.child_count[p];
        if (cn > 0 && i >= c0 && i < c0 + cn) count++;
    }
    return count;
}

static void apply_reflux(SimState *st) {

        int leaf = st->bank.leaf[i];
        if (leaf < 0 || leaf >= st->grid.n) continue;

    }
}

        return -2;
    }
    VaultHeader hdr;

        c->reason = REASON_SHORT;
        snprintf(c->msg, sizeof(c->msg), "header short");
        return -3;
    }
    if (hdr.magic != VAULT_MAGIC) {
        fclose(in);
        c->reason = REASON_MAGIC;
        snprintf(c->msg, sizeof(c->msg), "bad magic 0x%x", hdr.magic);
        return -4;
    }
    int n_cells = 0;

uint64_t vault_fnv1a(const void *data, int n_bytes, uint64_t seed) {
    const uint8_t *p = (const uint8_t *)data;
    uint64_t h = 0xcbf29ce484222325ull ^ seed;
    for (int i = 0; i < n_bytes; i++) {
        h ^= (uint64_t)p[i];
        h *= 0x100000001b3ull;

}
