static void mkpath(const char *path) {
    char buf[512];
    snprintf(buf, sizeof(buf), "%s", path);
    int n = (int)strlen(buf);
    for (int i = 1; i < n; i++) {
        if (buf[i] == '/') {
            buf[i] = 0;
            mkdir(buf, 0755);
            buf[i] = '/';

    }
}
