#include "../include/cell_view.h"

typedef void (*cell_fn)(CellView *, void *);

void traverse_t(CellGrid *a, cell_fn fn, void *ctx) {
    if (!a || !fn) return;
    for (int i = 0; i < a->n; i++) {
        fn(&a->cells[i], ctx);
    }
}

double span_t(const CellGrid *a) {
    if (!a || a->n == 0) return 0.0;
    double lo = a->cells[0].vol;
    double hi = a->cells[0].vol;
    for (int i = 1; i < a->n; i++) {
        double v = a->cells[i].vol;
        if (v < lo) lo = v;
        if (v > hi) hi = v;
    }
    return hi - lo;
}

void fold_q(const CellPatch *a, int b, CellPatch *c) {
    if (!a || !c) return;
    int n = (a->n < b) ? a->n : b;
    if (n <= 0) {
        c->n = 0;
        return;
    }
    double sum_vol = 0.0;
    double sum_q[NVAR] = {0.0, 0.0, 0.0};
    for (int i = 0; i < n; i++) {
        sum_vol += a->vol[i];
        for (int v = 0; v < NVAR; v++) {
            sum_q[v] += a->q[i * NVAR + v];
        }
    }
    c->n = 1;
    c->vol[0] = sum_vol;
    double inv_n = 1.0 / (double)n;
    for (int v = 0; v < NVAR; v++) {
        c->q[v] = sum_q[v] * inv_n;
