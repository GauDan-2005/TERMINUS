static void rstrip(char *s) {
    int n = (int)strlen(s);
    while (n > 0 && (s[n-1] == '\n' || s[n-1] == '\r' || s[n-1] == ' ' || s[n-1] == '\t')) {
        s[--n] = 0;
    }
}

static void parse_leaves(const char *val, ScenarioSpec *out) {
    int c = 0;
    const char *p = val;
    while (*p && c < MAX_COARSE) {
        int w = 0;
        int k = 0;
        while (*p && *p != '|') {
            if (*p >= '0' && *p <= '9') {
                w = w * 10 + (*p - '0');
            } else if (*p == ',') {
                if (k < MAX_LEAVES) out->leaf_weight[c][k++] = w;
                w = 0;
            }
            p++;
        }
        if (k < MAX_LEAVES) out->leaf_weight[c][k++] = w;
        out->leaf_count[c] = k;
        c++;
        if (*p == '|') p++;
    }
}
