static void esc(FILE *fp, const char *s) {
    fputc('"', fp);
    for (const char *p = s; *p; p++) {
        if (*p == '"' || *p == '\\') {
            fputc('\\', fp);
        }
        fputc(*p, fp);
    }
    fputc('"', fp);
}

    double s = 0.0;
    for (int i = 0; i < n; i++) {
        s += v[i] * v[i];
    }
    return sqrt(s);
