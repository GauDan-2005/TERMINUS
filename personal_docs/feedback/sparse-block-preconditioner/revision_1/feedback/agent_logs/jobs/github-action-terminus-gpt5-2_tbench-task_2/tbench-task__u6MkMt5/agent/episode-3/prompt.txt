    char path[512];
    CaseSpec spec;
    snprintf(path, sizeof(path), "/app/data/cases/%s.case", name);
    if (case_load(path, &spec) != 0) {
        return -1;
    }
    return run_u(&spec, out_dir, res);
}

        for (int i = 0; CASES[i]; i++) {

                return 1;
