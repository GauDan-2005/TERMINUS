    int dim = pack->n;

        y[i] = 0.0;
    }
    for (int bi = 0; bi < nb; bi++) {
        int start = pack->row_ptr[bi];
        int end = pack->row_ptr[bi + 1];
        for (int k = start; k < end; k++) {
            int bj = pack->col_idx[k];
            for (int r = 0; r < bs; r++) {
                for (int c = 0; c < bs; c++) {
                    int gi = bi * bs + r;
                    int gj = bj * bs + c;
                    y[gi] += pack->data[k * bs * bs + r * bs + c] * x[gj];
                }
            }
        }
    }
}
