function hasCargo() {

  return probe.status === 0;
}

function callNative(root, rev, primary, carry) {
  const args = [rev, String(primary), String(carry)];
  const proc = hasCargo()
    ? spawnSync('cargo', ['run', '--quiet', '--bin', 'mhre_native', '--', ...args], {
        cwd: root,
        encoding: 'utf8'
      })
    : spawnSync(`${root}/target/debug/mhre_native`, args, {
        cwd: root,
        encoding: 'utf8'
      });

  if (proc.status !== 0) {
    throw new Error(proc.stderr || proc.stdout || 'native worker failed');
  }
  return Number(proc.stdout.trim());
}
