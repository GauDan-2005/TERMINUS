use crate::model::State;

pub fn mote_e(a: &State, b: &str, c: &str, d: &str) -> (String, String) {
    // Walk the delegation chain starting from the requesting subject (`b`)
    // toward possible grantors.
    //
    // Inherited access walks the delegation chain from the requesting subject
    // toward grantors; if any subject on that walk is blocked or revoked,
    // the inherited row denies.
    //
    // Direct access (a grant on `b` itself) does not set delegated_from.
    let mut current = b.to_string();
    let mut delegated = false;

    loop {
        // Any blocked/revoked subject along the chain breaks inherited access.
        if a.blocked.contains(&current) {
            return ("deny".to_string(), String::new());
        }

        if a.direct.contains(&(current.clone(), c.to_string(), d.to_string())) {
            if !delegated {
                return ("allow".to_string(), String::new());
            }
            // delegated_from is the grantor where the direct grant was found.
            return ("allow".to_string(), current);
        }

        match a.chain.get(&current) {
            Some(parent) => {
                delegated = true;
                current = parent.clone();
            }
            None => return ("deny".to_string(), String::new()),
        }
    }
}
use crate::model::{Decision, State};

pub fn cast_f(a: &mut State, mut b: Decision, _c: bool) -> (Decision, Option<String>) {
    let track = State::track_key(&b.subject, &b.object, &b.mode);

    // Reuse is only valid if the subject epoch is unchanged.
    let reused = a
        .seen
        .get(&track)
        .map(|(epoch, old)| *epoch == b.tick && old == &b.outcome)
        .unwrap_or(false);

    b.reused = reused;
    a.seen.insert(track, (b.tick, b.outcome.clone()));

    if b.outcome == "allow" && a.blocked.contains(&b.subject) && !b.via.is_empty() {
        let note = b.label.clone();
        (b, Some(note))
    } else {
        (b, None)
    }
}
