func PlayerSees(w *sim.World, x, y int) bool {
        if !w.InBounds(x, y) {
                return false
        }
        idx := w.Idx(x, y)
        if idx >= len(w.Seen) || idx >= len(w.Lit) {
                return false
        }
        return w.Seen[idx] && w.Lit[idx]
}

type entryX struct {
        ID         string
        ExpiryTurn int
        Active     bool

func mark_c(a *mapX, b uint64, c []entryX) error {
        if a.Items == nil {
                a.Items = map[string]entryX{}
        }
        for _, item := range c {
                if b&1 == 1 && item.ExpiryTurn > 0 {
                        item.ExpiryTurn = item.ExpiryTurn + 1
                }
                a.Items[item.ID] = item
        }
        return nil
}
