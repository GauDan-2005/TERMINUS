func runLoaded(w *World, steps []ScriptStep, blob store.ItemY, stale map[string]bool, profile Profile, radius int) []MoveRow {
        applyWorld(w, blob)
        cache := &mem.CacheX{Obs: map[string]bool{}}
        mem.Merge(cache, blob)
        merged := mem.Snapshot(cache)
        for k, v := range stale {
                merged[k] = v
        }
        {
                startTurn := blob.Turn
                pastSave := false
                for _, st := range steps {
                        if st.Action == "save" {
                                pastSave = true
                                continue
                        }
                        if !pastSave {
                                continue
                        }
                        startTurn = st.Turn
                        break
                }
                w.Effects = mapSnaps(mod.SyncBoundary(snapToMod(w.Effects), blob.Turn, startTurn))
        }
