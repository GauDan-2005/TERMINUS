func Seed(profile ProfileSnap, turn int) []EffectSnap {
        out := []EffectSnap{}
        if profile.Blind {

        }
        if profile.Epoch {
                out = append(out, EffectSnap{ID: "ward", ExpiryTurn: profile.CheckTurn + 2, Active: true})
        }
        return out
}

                })
        }
        return rows
