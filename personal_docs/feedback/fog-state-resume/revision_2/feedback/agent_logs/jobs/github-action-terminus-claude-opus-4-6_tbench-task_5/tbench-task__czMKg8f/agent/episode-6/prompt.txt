static int abs_i(int v) {

    if (!a || !a->grid) {
        return 0;
    }
    if (x0 == x1 && y0 == y1) {
        return 1;
    }
    int dx = x1 - x0;
    int dy = y1 - y0;
    int adx = abs_i(dx);
    int ady = abs_i(dy);
    if (adx + ady != 1) {
        if (adx == ady && adx == 1) {
            return 1;
        }
        return 0;
    }

        return 0;
    }

        return 0;
    }
    return 1;
}
