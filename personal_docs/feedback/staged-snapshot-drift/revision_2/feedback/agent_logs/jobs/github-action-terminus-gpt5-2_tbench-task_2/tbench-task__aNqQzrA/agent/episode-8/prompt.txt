                if row.Op == "remove" {
                        _ = os.Remove(target)
                        for key, value := range groups {
                                if value == target {
                                        delete(groups, key)
                                }
                        }
                        continue
                }
                if err := os.MkdirAll(filepath.Dir(target), 0o755); err != nil {
                        return err

                key := d3.KeyFor(row, profile)
                if existing, ok := groups[key]; ok {

                                return err
                        }

                }

                                return err
