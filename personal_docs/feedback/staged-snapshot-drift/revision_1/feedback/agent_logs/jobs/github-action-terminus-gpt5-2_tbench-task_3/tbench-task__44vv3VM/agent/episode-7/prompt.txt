func statIdentity(path string) (uint64, uint64, uint64, error) {
        info, err := os.Stat(path)
        if err != nil {
                return 0, 0, 0, err
        }
        st := info.Sys().(*syscall.Stat_t)
        return uint64(st.Dev), uint64(st.Ino), uint64(st.Nlink), nil
}
